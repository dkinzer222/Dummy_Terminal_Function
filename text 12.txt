  <script>
    const themeSwitcher = document.getElementById('theme-switcher');
    const themePopup = document.getElementById('theme-popup');
    const themeSwitcherMobile = document.querySelector('.theme-switcher-mobile');
    const themePopupMobile = document.querySelector('.theme-popup-mobile');

    let isDragging = false;
    let startX, startY, initialX, initialY;
    let currentX, currentY;

    themeSwitcher.addEventListener('mousedown', startDragging);
    document.addEventListener('mousemove', drag);
    document.addEventListener('mouseup', stopDragging);

    themeSwitcher.addEventListener('touchstart', startDragging);
    document.addEventListener('touchmove', drag);
    document.addEventListener('touchend', stopDragging);

    function startDragging(e) {
      isDragging = true;
      if (e.type === 'mousedown') {
        startX = e.clientX;
        startY = e.clientY;
      } else {
        startX = e.touches[0].clientX;
        startY = e.touches[0].clientY;
      }
      initialX = themeSwitcher.offsetLeft;
      initialY = themeSwitcher.offsetTop;
      themeSwitcher.classList.remove('smooth-transition');
    }

    function drag(e) {
      if (isDragging) {
        e.preventDefault();
        if (e.type === 'mousemove') {
          currentX = e.clientX;
          currentY = e.clientY;
        } else {
          currentX = e.touches[0].clientX;
          currentY = e.touches[0].clientY;
        }
        requestAnimationFrame(updatePosition);
      }
    }

    function updatePosition() {
      const deltaX = currentX - startX;
      const deltaY = currentY - startY;
      
      let newX = initialX + deltaX;
      let newY = initialY + deltaY;
      
      const maxX = window.innerWidth - themeSwitcher.offsetWidth;
      const maxY = window.innerHeight - themeSwitcher.offsetHeight;
      
      newX = Math.max(0, Math.min(newX, maxX));
      newY = Math.max(0, Math.min(newY, maxY));
      
      themeSwitcher.style.left = `${newX}px`;
      themeSwitcher.style.top = `${newY}px`;
    }

    function stopDragging() {
      isDragging = false;
      themeSwitcher.classList.add('smooth-transition');
    }

    themeSwitcher.addEventListener('click', (e) => {
      if (!isDragging) {
        toggleThemePopup(themePopup, themeSwitcher);
      }
    });

    themeSwitcherMobile.addEventListener('click', () => {
      toggleThemePopup(themePopupMobile, themeSwitcherMobile);
    });

    function toggleThemePopup(popup, switcher) {
      popup.style.display = popup.style.display === 'block' ? 'none' : 'block';
      if (popup === themePopup) {
        popup.style.left = `${switcher.offsetLeft}px`;
        popup.style.top = `${switcher.offsetTop + switcher.offsetHeight + 10}px`;
      }
    }

    document.querySelectorAll('.theme-option').forEach(option => {
      option.addEventListener('click', () => {
        setTheme(option.dataset.theme);
        themePopup.style.display = 'none';
        themePopupMobile.style.display = 'none';
      });
    });

        existingIframe.remove();

        iframe.className = 'theme-iframe';
        document.body.appendChild(iframe);
      }

      updateThemeStyles(theme);
    }

    function performPortScan() {
      const ipAddress = document.getElementById('portScannerInput').value;
      const resultDiv = document.getElementById('portScannerResult');

      resultDiv.innerHTML = '<p>Initiating Nmap-like port scan... This may take a while.</p>';

      const totalPorts = 65535;
      const batchSize = 1000; // Scan ports in batches to avoid overwhelming the browser
      let scannedPorts = 0;
      let openPorts = [];
      let filteredPorts = [];

      function scanPortBatch(startPort) {
        const endPort = Math.min(startPort + batchSize - 1, totalPorts);
        const portPromises = [];

        for (let port = startPort; port <= endPort; port++) {
          portPromises.push(scanPort(ipAddress, port));
        }

        Promise.all(portPromises).then((results) => {
          results.forEach(result => {
            if (result.status === 'open') openPorts.push(result);
            if (result.status === 'filtered') filteredPorts.push(result);
          });

          scannedPorts += results.length;
          updateProgress(scannedPorts, totalPorts);

          if (scannedPorts < totalPorts) {
            scanPortBatch(endPort + 1);
          } else {
            displayResults(ipAddress, openPorts, filteredPorts);
          }
        });
      }

      scanPortBatch(1);
    }

    function scanPort(ipAddress, port) {
      return new Promise((resolve) => {
        const socket = new WebSocket(`wss://${ipAddress}:${port}`);

        socket.onopen = () => {
          resolve({ port, status: 'open', service: getServiceName(port) });
          socket.close();
        };

        socket.onerror = () => {
          resolve({ port, status: 'closed', service: getServiceName(port) });
        };

        setTimeout(() => {
          if (socket.readyState === WebSocket.CONNECTING) {
            socket.close();
            resolve({ port, status: 'filtered', service: getServiceName(port) });
          }
        }, 5000);
      });
    }

    function updateProgress(scanned, total) {
      const resultDiv = document.getElementById('portScannerResult');
      const percentage = Math.floor((scanned / total) * 100);
      resultDiv.innerHTML = `<p>Scanning... ${percentage}% complete (${scanned}/${total} ports)</p>`;
    }

    function generateHash() {
      const input = document.getElementById('hashInput').value;
      const hashType = document.getElementById('hashType').value;
      const resultDiv = document.getElementById('hashResult');

      const simpleHash = (str) => {
        let hash = 0;
        for (let i = 0; i < str.length; i++) {
          const char = str.charCodeAt(i);
          hash = ((hash << 5) - hash) + char;
          hash = hash & hash; // Convert to 32-bit integer
        }
        return hash.toString(16);
      };

      const hash = simpleHash(input);
      resultDiv.innerHTML = `<p>${hashType.toUpperCase()} Hash: ${hash}</p>`;
    }
